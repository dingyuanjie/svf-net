# 1. SVF Attention 核心

`model/svf_attention.py`

```Python
import torch
import torch.nn as nn
import torch.nn.functional as F


class SVFAttention(nn.Module):
    """
    Structural Vital Force Attention v0.1

    Attention + Structural Potential Field
    """

    def __init__(
        self,
        dim,
        heads=8,
        lambda_svf=0.1
    ):
        super().__init__()

        self.dim = dim
        self.heads = heads
        self.lambda_svf = lambda_svf

        self.head_dim = dim // heads

        self.qkv = nn.Linear(
            dim,
            dim * 3
        )

        self.out = nn.Linear(
            dim,
            dim
        )


        # 可学习结构角
        self.theta = nn.Parameter(
            torch.randn(2048)
        )

        # 可学习结构半径
        self.radius = nn.Parameter(
            torch.linspace(
                0,
                1,
                2048
            )
        )


    def structural_field(self, n, device):

        theta = self.theta[:n]

        r = self.radius[:n]


        # 角度耦合
        angle = (
            theta[:,None]
            -
            theta[None,:]
        )


        angle_field = torch.cos(angle)


        # 半径耦合
        radius_field = torch.exp(
            -torch.abs(
                r[:,None]
                -
                r[None,:]
            )
        )


        S = (
            0.5 * angle_field
            +
            0.5 * radius_field
        )


        return S.to(device)



    def forward(self,x):

        B,N,C=x.shape


        qkv=self.qkv(x)

        q,k,v=qkv.chunk(
            3,
            dim=-1
        )


        q=q.reshape(
            B,
            N,
            self.heads,
            self.head_dim
        ).transpose(1,2)


        k=k.reshape(
            B,
            N,
            self.heads,
            self.head_dim
        ).transpose(1,2)


        v=v.reshape(
            B,
            N,
            self.heads,
            self.head_dim
        ).transpose(1,2)



        score=torch.matmul(
            q,
            k.transpose(-2,-1)
        )


        score/=self.head_dim**0.5


        # ======================
        # SVF核心
        # ======================

        S=self.structural_field(
            N,
            x.device
        )


        score = score + (
            self.lambda_svf
            *
            S
        )



        attn=F.softmax(
            score,
            dim=-1
        )


        out=torch.matmul(
            attn,
            v
        )


        out=out.transpose(
            1,2
        ).reshape(
            B,N,C
        )


        return self.out(out)
```

# 2. SVF Block

`model/svf_block.py`

```Python
import torch.nn as nn

from .svf_attention import SVFAttention


class SVFBlock(nn.Module):

    def __init__(
        self,
        dim,
        heads
    ):
        super().__init__()


        self.norm1=nn.LayerNorm(dim)

        self.attn=SVFAttention(
            dim,
            heads
        )


        self.norm2=nn.LayerNorm(dim)


        self.ffn=nn.Sequential(
            nn.Linear(
                dim,
                dim*4
            ),
            nn.GELU(),

            nn.Linear(
                dim*4,
                dim
            )
        )


    def forward(self,x):

        x=x+self.attn(
            self.norm1(x)
        )

        x=x+self.ffn(
            self.norm2(x)
        )

        return x
```

# 3. SVF Transformer

`model/svf_transformer.py`

```Python
import torch.nn as nn

from .svf_block import SVFBlock



class SVFTransformer(nn.Module):

    def __init__(
        self,
        vocab,
        dim=512,
        depth=6,
        heads=8,
        max_len=2048
    ):
        super().__init__()


        self.token_emb=nn.Embedding(
            vocab,
            dim
        )


        self.pos_emb=nn.Parameter(
            torch.randn(
                1,
                max_len,
                dim
            )
        )


        self.layers=nn.ModuleList(
            [
                SVFBlock(
                    dim,
                    heads
                )
                for _ in range(depth)
            ]
        )


        self.norm=nn.LayerNorm(dim)


        self.head=nn.Linear(
            dim,
            vocab
        )


    def forward(self,x):

        B,N=x.shape


        h=self.token_emb(x)

        h=h+self.pos_emb[:,:N]


        for layer in self.layers:

            h=layer(h)


        h=self.norm(h)


        return self.head(h)
```
